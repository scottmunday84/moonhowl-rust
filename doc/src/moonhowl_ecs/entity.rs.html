6pub struct Entity {
7    components: HashMap<TypeId, Box<dyn IComponent>>,
8    registered_components: HashMap<TypeId, HashSet<usize>>,
9}
10
11impl Entity {
12    pub fn new() -> Self {
13        Self {
14            components: HashMap::new(),
15            registered_components: HashMap::new(),
16        }
17    }
18
19    pub fn has_component<T: IComponent>(&self) -> bool {
20        self.components.contains_key(&TypeId::of::<T>())
21    }
22
23    pub fn get_component<T: IComponent>(&self) -> Option<&T> {
24        self.components
25            .get(&TypeId::of::<T>())
26            .and_then(|component| (**component).as_any().downcast_ref::<T>())
27    }
28
29    pub fn has_registered_component<T: IComponent>(&self, id: &usize) -> bool {
30        if !self.has_component::<T>() {
31            return false;
32        }
33
34        if let Some(registered_component) = self.registered_components.get(&TypeId::of::<T>()) {
35            return !registered_component.contains(id);
36        }
37
38        true
39    }
40
41    pub fn get_registered_component_readonly<T: IComponent>(&self, id: &usize) -> Option<&T> {
42        if !self.has_registered_component::<T>(id) {
43            return None;
44        }
45
46        self.get_component::<T>()
47    }
48
49    pub fn get_registered_component<T: IComponent>(&mut self, id: &usize) -> Option<&T> {
50        if !self.has_registered_component::<T>(id) {
51            return None;
52        }
53
54        if let Some(registered_component) = self.registered_components.get_mut(&TypeId::of::<T>()) {
55            registered_component.insert(*id);
56        };
57
58        self.get_component::<T>()
59    }
60
61    pub fn push_component<T: IComponent>(&mut self, component: T) -> &mut Self {
62        let boxed_component = Box::new(component);
63        self.components.insert(TypeId::of::<T>(), boxed_component);
64        self.registered_components
65            .insert(TypeId::of::<T>(), HashSet::new());
66
67        self
68    }
69
70    pub fn pull_component<T: IComponent>(&mut self) -> &mut Self {
71        self.components.remove(&TypeId::of::<T>());
72        self.registered_components.remove(&TypeId::of::<T>());
73
74        self
75    }
76
77    pub fn check<F>(&self, predicate: F) -> EntityCheck
78    where
79        F: FnOnce(&Entity) -> bool,
80    {
81        EntityCheck(predicate(&self))
82    }
83}

85pub enum EntitySystem<'a> {
86    Reader(&'a Entity, &'a System),
87    Writer(&'a mut Entity, &'a System),
88}
89
90impl<'a> EntitySystem<'a> {
91    pub fn new_reader(entity: &'a Entity, system: &'a System) -> Self {
92        Self::Reader(entity, system)
93    }
94
95    pub fn new_writer(entity: &'a mut Entity, system: &'a System) -> Self {
96        Self::Writer(entity, system)
97    }
98
99    pub fn has_component<T: IComponent>(&self) -> bool {
100        match self {
101            EntitySystem::Reader(entity, system) => system.has_component::<T>(entity),
102            EntitySystem::Writer(entity, system) => system.has_component::<T>(entity),
103        }
104    }
105
106    pub fn get_component<T: IComponent>(&mut self) -> Option<&T> {
107        match self {
108            EntitySystem::Reader(entity, system) => system.get_component_readonly::<T>(entity),
109            EntitySystem::Writer(entity, system) => system.get_component::<T>(entity),
110        }
111    }
112
113    pub fn check<F>(&self, predicate: F) -> EntityCheck
114    where
115        F: FnOnce(&EntitySystem) -> bool,
116    {
117        EntityCheck(predicate(&self))
118    }
119}
